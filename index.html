    "id": "q1",
    "subject": "数学",
    "type": "choice",
    "title": "选择正确答案",
    "answer": "B",
    "score": 2
  },
  {
    "id": "q2",
    "subject": "语文",
    "type": "essay",
    "title": "那一刻，我长大了",
    "score": 50,
    "keywords": ["长大", "责任", "感受"],
    "rubric": ["中心明确", "事例具体", "结尾点题"]
  }
]

学生作业 JSON

[
  {
    "student": "李同学",
    "answers": {
      "q1": "B",
      "q2": "我照顾生病的妈妈，第一次明白责任。"
    }
  }
]
